    ① trend_scout.py — Find what's viral right now
    PYTHON 
    from pytrends.request import TrendReq

def get_trending_topics(niche="ASMR"):
    pytrends = TrendReq(hl='en-US', tz=360)
    pytrends.build_payload(
        [niche], cat=0,
        timeframe='now 1-d',  # last 24 hours
        geo=''
    )
    related = pytrends.related_queries()
    # Extract top rising queries
    rising = related[niche]['rising']
    return rising['query'].tolist()[:5]

# Example output: ['asmr rain sounds', 'asmr sleep 2025', ...]
    ② script_gen.py — AI writes your video script
    PYTHON 
    import ollama

def generate_script(topic: str) -> dict:
    prompt = f"""
    Create a viral YouTube Shorts video plan for: {topic}
    
    Return EXACTLY this JSON format:
    {{
      "title": "clickbait YouTube title under 60 chars",
      "description": "SEO description with keywords",
      "hashtags": ["tag1", "tag2", "tag3"],
      "scenes": [
        {{"narration": "text to speak", "visual": "image description for AI"}}
      ]
    }}
    Keep scenes between 3-5. Make it engaging and trending.
    """
    
    response = ollama.chat(
        model='mistral',
        messages=[{'role': 'user', 'content': prompt}]
    )
    import json
    return json.loads(response['message']['content'])
    ③ image_gen.py — Generate visuals via Stable Diffusion API
    PYTHON 
    import requests, base64
from PIL import Image
import io

# AUTOMATIC1111 must be running with --api flag
# Start it with: python launch.py --api --xformers
SD_URL = "http://127.0.0.1:7860"

def generate_image(prompt: str, output_path: str):
    payload = {
        "prompt": prompt + ", 4k, cinematic, trending",
        "negative_prompt": "blurry, ugly, watermark",
        "steps": 20,        # Fast on RTX 4050
        "width": 576,        # Vertical (9:16 for Shorts)
        "height": 1024,
        "cfg_scale": 7,
        "sampler_name": "DPM++ 2M Karras"
    }
    r = requests.post(f"{SD_URL}/sdapi/v1/txt2img", json=payload)
    img_data = base64.b64decode(r.json()['images'][0])
    Image.open(io.BytesIO(img_data)).save(output_path)
    return output_path
    ④ voice_gen.py — AI voice / ASMR narration
    PYTHON 
    from bark import SAMPLE_RATE, generate_audio, preload_models
from scipy.io.wavfile import write as write_wav
import numpy as np

preload_models()  # Downloads models once, runs on GPU

def generate_voiceover(text: str, output_path: str,
                        asmr_mode=False):
    # Bark voice presets: v2/en_speaker_0 through _9
    voice = "v2/en_speaker_6"
    
    if asmr_mode:
        # Add whisper tags for ASMR effect
        text = f"[whisper]{text}[/whisper]"
    
    audio_array = generate_audio(text, history_prompt=voice)
    write_wav(output_path, SAMPLE_RATE, audio_array)
    return output_path
    ⑤ video_composer.py — Assemble the final video
    PYTHON 
    from moviepy.editor import (
    ImageClip, AudioFileClip, concatenate_videoclips,
    TextClip, CompositeVideoClip
)

def compose_video(scenes: list, output_path: str):
    clips = []
    
    for i, scene in enumerate(scenes):
        # Load image and audio for each scene
        audio = AudioFileClip(scene['audio'])
        img = (ImageClip(scene['image'])
               .set_duration(audio.duration)
               .set_audio(audio))
        
        # Add caption text overlay
        txt = (TextClip(scene['narration'], fontsize=40,
                       color='white', font='Arial-Bold',
                       size=(540, None), method='caption')
               .set_duration(audio.duration)
               .set_pos(('center', 'bottom')))
        
        clips.append(CompositeVideoClip([img, txt]))
    
    final = concatenate_videoclips(clips)
    final.write_videofile(output_path, fps=30, codec='libx264')
    return output_path
    ⑥ uploader.py — Auto-post to YouTube & Instagram
    PYTHON 
    # ─── YouTube Upload ───────────────────────────
from googleapiclient.discovery import build
from googleapiclient.http import MediaFileUpload
from google.oauth2.credentials import Credentials

def upload_to_youtube(video_path, title, description, tags):
    creds = Credentials.from_authorized_user_file('token.json')
    yt = build('youtube', 'v3', credentials=creds)
    
    request = yt.videos().insert(
        part="snippet,status",
        body={
            "snippet": {
                "title": title,
                "description": description,
                "tags": tags,
                "categoryId": "22"  # People & Blogs
            },
            "status": {"privacyStatus": "public"}
        },
        media_body=MediaFileUpload(video_path)
    )
    return request.execute()

# ─── Instagram Upload ─────────────────────────
from instagrapi import Client

def upload_to_instagram(video_path, caption, hashtags):

    ⑦ agent.py — The main orchestrator (run this!)
    PYTHON — main agent loop 
    from trend_scout import get_trending_topics
from script_gen import generate_script
from image_gen import generate_image
from voice_gen import generate_voiceover
from video_composer import compose_video
from uploader import upload_to_youtube, upload_to_instagram
from apscheduler.schedulers.blocking import BlockingScheduler
import os, time

NICHES = ["ASMR", "baby videos", "viral challenge", "satisfying videos"]

def run_agent():
    niche = NICHES[int(time.time()) % len(NICHES)]
    print(f"🔍 Scouting trends for: {niche}")

    topics = get_trending_topics(niche)
    topic = topics[0]  # Pick #1 trending
    print(f"📈 Top trend: {topic}")

    plan = generate_script(topic)
    print(f"✍️  Script ready: {plan['title']}")

    scenes_data = []
    for i, scene in enumerate(plan['scenes']):
        img = generate_image(scene['visual'], f"scene_{i}.png")
        audio = generate_voiceover(scene['narration'], f"audio_{i}.wav",
                                   asmr_mode=(niche=="ASMR"))
        scenes_data.append({'image': img, 'audio': audio,
                             'narration': scene['narration']})

    video = compose_video(scenes_data, "output/final.mp4")
    print("🎬 Video composed!")

    upload_to_youtube(video, plan['title'],
                      plan['description'], plan['hashtags'])
    upload_to_instagram(video, plan['description'], plan['hashtags'])
    print("🚀 Posted to YouTube & Instagram!")

# ─── Schedule: run 3x daily automatically ─────
scheduler = BlockingScheduler()
scheduler.add_job(run_agent, 'cron', hour='8,14,20')
print("⏰ Agent running 3x daily. Press Ctrl+C to stop.")
scheduler.start()

    PRO TIPS Maximize Viral Potential
    🎯 Hook in First 3 Seconds
    Tell your LLM to always start narration with a shocking question or statement. This is what stops the scroll on Shorts/Reels.
    📊 A/B Test Thumbnails
    Generate 3 different cover images per video with different prompts. Use YouTube Studio analytics after a week to see which style wins.
    ⏰ Post at Peak Hours
    Schedule posts at 8 AM, 2 PM, and 8 PM in your target audience's timezone. These are highest-engagement windows on both platforms.
    🔄 Rotate Niches Daily
    Don't post only ASMR. Rotate through ASMR → baby content → satisfying videos → challenges. This hedges against algorithm shifts.
    🎓
    Beginner Tip: Don't try to build everything at once. Start by running each module (trend_scout.py, script_gen.py, etc.) individually and verify it works before connecting them in agent.py. Debugging is much easier this way.
